SECURE_BANK.PY

class Vault:
    def __init__(self, bal): self.bal = bal
    def withdraw(self, amt):
        if amt <= self.bal: 
            self.bal -= amt
            return f"Success. New: {self.bal}"
        return "Insufficient Funds"

user = Vault(5000)
print(user.withdraw(1200))

DATA_SHIFT.PY

def encrypt(text, s):
    res = ""
    for char in text:
        res += chr((ord(char) + s - 65) % 26 + 65)
    return res

print(encrypt("UTKARSH", 4))
